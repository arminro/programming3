621  private void DrawBoard(DrawingContext dc, int indexer, Pen myPen)
622  {
623  // THE TRACK --> ALWAYS RENDERED AFTER THE BACKGROUND
624  // start column
625  ImageBrush brush;
626 
627  Rect myRect = new Rect(this.bl.StartX, this.bl.StartY, this.bl.Met.SquareCardMetric, this.bl.Met.SquareCardMetric);
628 
629  dc.DrawRectangle((ImageBrush)this.Resources["start"], myPen, myRect);
630 
631  // lower horizontal part
632  int changingWidth = this.bl.Met.NormalCardWidth;
633  int correction = 0;
634  Point holder; // this holds the calcuated points
635  for (int i = 1; i <= this.bl.Met.NumberOfElementsInAHorizontalRow; i++)
636  {
637  // if we reached the last in the row (and it cannot be 0 bc of the starting idx)
638  brush = (ImageBrush)this.Resources[this.bl.GameBoard[indexer].ImageKey];
639  if (i % this.bl.Met.NumberOfElementsInAHorizontalRow == 0)
640  {
641  changingWidth = this.bl.Met.SquareCardMetric;
642 
643  // we have to put the square card a little further, because it is wider
644  correction = this.bl.Met.SquareCardMetric - this.bl.Met.NormalCardWidth;
645  }
646 
647  holder = new Point(this.bl.StartX - (i * this.bl.Met.NormalCardWidth) - correction, this.bl.StartY);
648  this.bl.GameBoard[indexer].Rect = this.bl.CalculatePrimaryPosition(holder, changingWidth, this.bl.Met.NormalCardHeight);
649  myRect = new Rect(holder.X, holder.Y, changingWidth, this.bl.Met.NormalCardHeight);
650  dc.DrawRectangle(brush, myPen, myRect);
651 
652  // drawing the number of the card, we treat square cards differently, as they do not have the number on their edges
653  Vector numberVector;
654  if (this.bl.CardIsSquare(indexer))
655  {
656  numberVector = new Vector(-40, 60);
657  }
658  else
659  {
660  numberVector = new Vector(0, 60);
661  }
662 
663  this.DrawText(dc, Point.Subtract(this.bl.GameBoard[indexer].Rect, numberVector), indexer.ToString());
664  indexer++;
665  }
666 
667  // left vertical part
668  changingWidth = this.bl.Met.NormalCardWidth;
669  correction = 0;
670 
671  /*The vertical parts are tricky, because we want to use images here that are tilted by 90° */
672  for (int i = 1; i <= this.bl.Met.NumberOfElementsInAVerticalRow; i++)
673  {
674  // save the imagekey of the gameboard
675  string temp = this.bl.GameBoard[indexer].ImageKey;
676 
677  // we check if the image is not among the square ones, as they do not have tilted alternatives
678  if (this.bl.ResourceNamesSquare.Contains(temp))
679  {
680  brush = (ImageBrush)this.Resources[temp];
681  }
682  else
683  {
684  // the string[]s for both the normal and tilted images have the same order
685  // so we we can look for the index of temp in the string[] of normal images
686  // and use its index to access the name of the tilted image
687  int indexInHorizontal = Array.FindIndex(this.bl.ResourceNamesNormal, p => p == temp);
688  brush = (ImageBrush)this.Resources[this.bl.ResourceNamesHorizontal[indexInHorizontal]];
689  }
690 
691  if (i % this.bl.Met.NumberOfElementsInAVerticalRow == 0)
692  {
693  changingWidth = this.bl.Met.SquareCardMetric;
694 
695  // we have to put the square card a little further, because it is wider
696  correction = this.bl.Met.SquareCardMetric - this.bl.Met.NormalCardWidth;
697  }
698 
699  holder = new Point(this.bl.LeftVerticalAlign, (this.bl.StartY - (i * this.bl.Met.NormalCardWidth)) - correction);
700  this.bl.GameBoard[indexer].Rect = this.bl.CalculatePrimaryPosition(holder, this.bl.Met.NormalCardHeight, changingWidth);
701  myRect = new Rect(holder.X, holder.Y, this.bl.Met.NormalCardHeight, changingWidth);
702  dc.DrawRectangle(brush, myPen, myRect);
703 
704  // drawing the number of the card
705  Vector numberVector;
706  if (this.bl.CardIsSquare(indexer))
707  {
708  numberVector = new Vector(-42, -40);
709  }
710  else
711  {
712  numberVector = new Vector(-43, 15);
713  }
714 
715  this.DrawText(dc, Point.Subtract(this.bl.GameBoard[indexer].Rect, numberVector), indexer.ToString());
716  indexer++;
717  }
718 
719  // upper horizontal part
720  changingWidth = this.bl.Met.NormalCardWidth;
721  correction = 0;
722  for (int i = 1; i <= this.bl.Met.NumberOfElementsInAHorizontalRow; i++)
723  {
724  // if we reached the last in the row (and it cannot be 0 bc of the starting idx)
725  brush = (ImageBrush)this.Resources[this.bl.GameBoard[indexer].ImageKey];
726  if (i % this.bl.Met.NumberOfElementsInAHorizontalRow == 0)
727  {
728  changingWidth = this.bl.Met.SquareCardMetric;
729 
730  // we have to put the square card a little further, because it is wider
731  correction = this.bl.Met.SquareCardMetric - this.bl.Met.NormalCardWidth;
732  }
733 
734  holder = new Point(
735  (this.bl.LeftVerticalAlign + (i * this.bl.Met.NormalCardWidth))
736  + this.bl.Met.SquareCardMetric - this.bl.Met.NormalCardWidth, this.bl.UpperVerticalAlign);
737  this.bl.GameBoard[indexer].Rect = this.bl.CalculatePrimaryPosition(holder, changingWidth, this.bl.Met.NormalCardHeight);
738 
739  myRect = new Rect(holder.X, holder.Y, changingWidth, this.bl.Met.NormalCardHeight);
740  dc.DrawRectangle(brush, myPen, myRect);
741 
742  // drawing the number of the card
743  Vector numberVector;
744  if (this.bl.CardIsSquare(indexer))
745  {
746  numberVector = new Vector(60, -40);
747  }
748  else
749  {
750  numberVector = new Vector(10, -40);
751  }
752 
753  this.DrawText(dc, Point.Subtract(this.bl.GameBoard[indexer].Rect, numberVector), indexer.ToString());
754  indexer++;
755  }
756 
757  changingWidth = this.bl.Met.NormalCardWidth;
758  correction = 0;
759  for (int i = 1; i <= this.bl.Met.NumberOfElementsInAVerticalRow - 1; i++)
