284  private void DrawBoard(DrawingContext dc, int indexer, Pen myPen)
285  {
286  // THE TRACK --> ALWAYS RENDERED AFTER THE BACKGROUND
287  // start column
288  ImageBrush brush;
289 
290  Rect myRect = new Rect(this.bl.StartX, this.bl.StartY, this.bl.Met.SquareCardMetric, this.bl.Met.SquareCardMetric);
291 
292  dc.DrawRectangle((ImageBrush)this.Resources["start"], myPen, myRect);
293 
294  // lower horizontal part
295  int changingWidth = this.bl.Met.NormalCardWidth;
296  int correction = 0;
297  Point holder; // this holds the calcuated points
298  for (int i = 1; i <= this.bl.Met.NumberOfElementsInAHorizontalRow; i++)
299  {
300  // if we reached the last in the row (and it cannot be 0 bc of the starting idx)
301  brush = (ImageBrush)this.Resources[this.bl.GameBoard[indexer].ImageKey];
302  if (i % this.bl.Met.NumberOfElementsInAHorizontalRow == 0)
303  {
304  changingWidth = this.bl.Met.SquareCardMetric;
305 
306  // we have to put the square card a little further, because it is wider
307  correction = this.bl.Met.SquareCardMetric - this.bl.Met.NormalCardWidth;
308  }
309 
310  holder = new Point(this.bl.StartX - (i * this.bl.Met.NormalCardWidth) - correction, this.bl.StartY);
311  this.bl.GameBoard[indexer].Rect = this.bl.CalculatePrimaryPosition(holder, changingWidth, this.bl.Met.NormalCardHeight);
312  myRect = new Rect(holder.X, holder.Y, changingWidth, this.bl.Met.NormalCardHeight);
313  dc.DrawRectangle(brush, myPen, myRect);
314 
315  // drawing the number of the card, we treat square cards differently, as they do not have the number on their edges
316  Vector numberVector;
317  if (this.bl.CardIsSquare(indexer))
318  {
319  numberVector = new Vector(-40, 60);
320  }
321  else
322  {
323  numberVector = new Vector(0, 60);
324  }
325 
326  this.DrawText(dc, Point.Subtract(this.bl.GameBoard[indexer].Rect, numberVector), indexer.ToString());
327  indexer++;
328  }
329 
330  // left vertical part
331  changingWidth = this.bl.Met.NormalCardWidth;
332  correction = 0;
333 
334  /*The vertical parts are tricky, because we want to use images here that are tilted by 90° */
335  for (int i = 1; i <= this.bl.Met.NumberOfElementsInAVerticalRow; i++)
336  {
337  // save the imagekey of the gameboard
338  string temp = this.bl.GameBoard[indexer].ImageKey;
339 
340  // we check if the image is not among the square ones, as they do not have tilted alternatives
341  if (this.bl.ResourceNamesSquare.Contains(temp))
342  {
343  brush = (ImageBrush)this.Resources[temp];
344  }
345  else
346  {
347  // the string[]s for both the normal and tilted images have the same order
348  // so we we can look for the index of temp in the string[] of normal images
349  // and use its index to access the name of the tilted image
350  int indexInHorizontal = Array.FindIndex(this.bl.ResourceNamesNormal, p => p == temp);
351  brush = (ImageBrush)this.Resources[this.bl.ResourceNamesHorizontal[indexInHorizontal]];
352  }
353 
354  if (i % this.bl.Met.NumberOfElementsInAVerticalRow == 0)
355  {
356  changingWidth = this.bl.Met.SquareCardMetric;
357 
358  // we have to put the square card a little further, because it is wider
359  correction = this.bl.Met.SquareCardMetric - this.bl.Met.NormalCardWidth;
360  }
361 
362  holder = new Point(this.bl.LeftVerticalAlign, (this.bl.StartY - (i * this.bl.Met.NormalCardWidth)) - correction);
363  this.bl.GameBoard[indexer].Rect = this.bl.CalculatePrimaryPosition(holder, this.bl.Met.NormalCardHeight, changingWidth);
364  myRect = new Rect(holder.X, holder.Y, this.bl.Met.NormalCardHeight, changingWidth);
365  dc.DrawRectangle(brush, myPen, myRect);
366 
367  // drawing the number of the card
368  Vector numberVector;
369  if (this.bl.CardIsSquare(indexer))
370  {
371  numberVector = new Vector(-42, -40);
372  }
373  else
374  {
375  numberVector = new Vector(-43, 15);
376  }
377 
378  this.DrawText(dc, Point.Subtract(this.bl.GameBoard[indexer].Rect, numberVector), indexer.ToString());
379  indexer++;
380  }
381 
382  // upper horizontal part
383  changingWidth = this.bl.Met.NormalCardWidth;
384  correction = 0;
385  for (int i = 1; i <= this.bl.Met.NumberOfElementsInAHorizontalRow; i++)
386  {
387  // if we reached the last in the row (and it cannot be 0 bc of the starting idx)
388  brush = (ImageBrush)this.Resources[this.bl.GameBoard[indexer].ImageKey];
389  if (i % this.bl.Met.NumberOfElementsInAHorizontalRow == 0)
390  {
391  changingWidth = this.bl.Met.SquareCardMetric;
392 
393  // we have to put the square card a little further, because it is wider
394  correction = this.bl.Met.SquareCardMetric - this.bl.Met.NormalCardWidth;
395  }
396 
397  holder = new Point(
398  (this.bl.LeftVerticalAlign + (i * this.bl.Met.NormalCardWidth))
399  + this.bl.Met.SquareCardMetric - this.bl.Met.NormalCardWidth, this.bl.UpperVerticalAlign);
400  this.bl.GameBoard[indexer].Rect = this.bl.CalculatePrimaryPosition(holder, changingWidth, this.bl.Met.NormalCardHeight);
401 
402  myRect = new Rect(holder.X, holder.Y, changingWidth, this.bl.Met.NormalCardHeight);
403  dc.DrawRectangle(brush, myPen, myRect);
404 
405  // drawing the number of the card
406  Vector numberVector;
407  if (this.bl.CardIsSquare(indexer))
408  {
409  numberVector = new Vector(60, -40);
410  }
411  else
412  {
413  numberVector = new Vector(10, -40);
414  }
415

417  indexer++;
418  }

421  correction = 0;

423  {

444  this.bl.GameBoard[indexer].Rect = this.bl.CalculatePrimaryPosition(holder, this.bl.Met.NormalCardHeight, changingWidth);
